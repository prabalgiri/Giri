const fs = require('fs');

const dataToStore = {
  name: 'John Doe',
  age: 25,
  city: 'Example City',
};

const jsonData = JSON.stringify(dataToStore, null, 2); // The third argument (2) is for indentation in JSON.stringify

const filePath = 'data.json';

fs.writeFile(filePath, jsonData, (err) => {
  if (err) {
    console.error('Error writing to file:', err);
  } else {
    console.log('Data has been written to the file successfully!');
  }
});
