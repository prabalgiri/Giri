document.addEventListener('DOMContentLoaded', function() {
  // Attach event listeners to all delete buttons
  const deleteButtons = document.querySelectorAll('.deleteButton');
  
  deleteButtons.forEach(button => {
    button.addEventListener('click', function() {
      const userId = this.getAttribute('data-user-id');
      
      // Make a DELETE request to your server
      axios.delete(`https://your-api-endpoint/users/${userId}`)
        .then(response => {
          // On successful deletion, remove the user detail from the website
          const userDetailItem = this.closest('li');
          userDetailItem.remove();
          
          console.log('User detail deleted:', response.data);
        })
        .catch(error => {
          console.error('Error deleting user detail:', error.message);
        });
    });
  });
});
